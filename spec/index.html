function DefineEnumerationType(target, keys, values) {
  let descriptors = {};

  for (let key of keys) {
    let key = keys[k];
    let hasKey = values.hasOwnProperty(key);
    let value = hasKey ? values[key] : Symbol(key);
    descriptors[key] = {
      value,
      configurable: false,
      writable: false,
      enumerable: true,
    };
  }

  Object.defineProperties(target, descriptors);
}


// enum expression
let Days = enum {
  SUNDAY,
  MONDAY,
  TUESDAY,
  WEDNESDAY,
  THURSDAY,
  FRIDAY,
  SATURDAY,
};

// Desugars to:
let Days;
{
  Days = Object.create(null);
  let keys = [
    "SUNDAY",
    "MONDAY",
    "TUESDAY",
    "WEDNESDAY",
    "THURSDAY",
    "FRIDAY",
    "SATURDAY",
  ];
  let values = {};

  DefineEnumerationType(Days, keys, values);
}
console.log(Days.SUNDAY); // Symbol(SUNDAY)
console.log(Days.SATURDAY); // Symbol(SATURDAY)
